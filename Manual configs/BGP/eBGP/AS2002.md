## R-2002-A

```c
conf t
!
router bgp 2002
 bgp router-id 3.1.1.1
 bgp log-neighbor-changes
 !
 neighbor 130.0.3.5 remote-as 1000 !To R-1001-I
 neighbor 2221:2566:1:1130::1 remote-as 1000 !To R-1001-I
 !
 address-family ipv4
  redistribute static
  neighbor 130.0.3.5 activate
  exit-address-family
 !
 address-family ipv6
  redistribute static
  neighbor 2221:2566:1:1130::1 activate
  exit-address-family
 !
 exit
!
! STATIC ROUTES ON AS EXIT
!
ip route 132.0.0.0 255.255.0.0 Null0
ipv6 route 2221:2566:3::/48 Null0
!
end
!

```

## R-2002-C

```c
conf t
!
router bgp 2002
 bgp router-id 3.1.1.3
 bgp log-neighbor-changes
 !
 neighbor 130.0.3.1 remote-as 1000 !To R-1001-H
 neighbor 2221:2566:1:1030::1 remote-as 1000 !To R-1001-H
 !
 neighbor 133.0.255.1 remote-as 3001 !To R-3001-A
 neighbor 133.0.255.1 update-source loopback0 !To R-3001-A
 neighbor 133.0.255.1 ebgp-multihop !To R-3001-A
 !
 neighbor 134.0.255.1 remote-as 3002 !To R-3002-A
 neighbor 134.0.255.1 update-source loopback0 !To R-3002-A
 neighbor 134.0.255.1 ebgp-multihop !To R-3002-A
 !
 neighbor 2221:2566:4:3001::A remote-as 3001 !To R-3001-A
 neighbor 2221:2566:4:3001::A update-source loopback0 !To R-3001-A
 neighbor 2221:2566:4:3001::A ebgp-multihop !To R-3001-A
 !
 neighbor 2221:2566:5:3002::A remote-as 3002 !To R-3002-A
 neighbor 2221:2566:5:3002::A update-source loopback0 !To R-3002-A
 neighbor 2221:2566:5:3002::A ebgp-multihop !To R-3002-A
 !
 address-family ipv4
  redistribute static
  neighbor 130.0.3.1 activate
  neighbor 133.0.255.1 activate
  neighbor 134.0.255.1 activate
  exit-address-family
 !

  redistribute static
  neighbor 2221:2566:1:1030::1 activate
  neighbor 2221:2566:4:3001::A activate

 !
 exit
!

!
```
